def initialize(attributes = {})
  # Set defaults
  @model = 'array'
  
  super(attributes)
  
  # Convert array to Numo::DFloat
  if @array.is_a?(Array)
    @array = Numo::DFloat.cast(@array)
  end
  
  validate!
end
